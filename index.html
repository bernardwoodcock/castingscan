<script>

    // Persistence Layer

    const fields = ['aiKey', 'searchKey', 'aiModel', 'pName', 'pAge', 'pLoc', 'pCredits', 'pSkills'];

    window.onload = () => {

        fields.forEach(f => {

            const val = localStorage.getItem('cast_' + f);

            if (val) document.getElementById(f).value = val;

        });

    };


    function save() {

        fields.forEach(f => localStorage.setItem('cast_' + f, document.getElementById(f).value));

    }


    function toggleSettings() {

        const d = document.getElementById('settingsDrawer');

        d.style.display = d.style.display === 'block' ? 'none' : 'block';

    }


    function log(msg) {

        const c = document.getElementById('console');

        c.style.display = 'block';

        const entry = document.createElement('div');

        entry.textContent = `> ${new Date().toLocaleTimeString()}: ${msg}`;

        c.prepend(entry);

    }


    async function runScout() {

        save();

        const aiKey = document.getElementById('aiKey').value;

        const sKey = document.getElementById('searchKey').value;

        const btn = document.getElementById('scanBtn');


        if (!aiKey || !sKey) return alert("Please provide both API keys in settings.");

        

        btn.disabled = true;

        document.getElementById('results').innerHTML = '';

        

        try {

            log("Opening live search socket via Tavily...");

            const searchQuery = `Current Australian casting calls ${document.getElementById('pAge').value} ${document.getElementById('sKeys').value} 2026`;

            

            const searchResp = await fetch('https://api.tavily.com/search', {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({ api_key: sKey, query: searchQuery, search_depth: "advanced", max_results: 5 })

            });

            const searchData = await searchResp.json();

            log(`Found ${searchData.results.length} potential sources. Filtering...`);


            const rawContent = searchData.results.map(r => `SOURCE: ${r.url}\nCONTENT: ${r.content}`).join("\n\n");


            log("Analyzing data with LLM Agent...");

            const profile = `Name: ${document.getElementById('pName').value}, Age: ${document.getElementById('pAge').value}, Loc: ${document.getElementById('pLoc').value}, Credits: ${document.getElementById('pCredits').value}`;

            

            const results = await callAI(aiKey, profile, rawContent);

            renderResults(results);

            log("Scan complete.");


        } catch (e) {

            log("ERROR: " + e.message);

            alert("Scanner failure. Check console for details.");

        } finally {

            btn.disabled = false;

        }

    }


    async function callAI(key, profile, content) {

        const model = document.getElementById('aiModel').value;

        const prompt = `You are an expert talent agent. Match this profile: [${profile}] against this live web data: [${content}]. 

        Return a JSON object with a key "opportunities" containing an array of 5 matches. Each match needs: title, type, location, matchScore (0-100), link, and a 1-sentence "Why" explaining the fit based on the actor's specific credits or age.`;


        if (model === 'anthropic') {

            const resp = await fetch('https://api.anthropic.com/v1/messages', {

                method: 'POST',

                headers: { 'x-api-key': key, 'anthropic-version': '2023-06-01', 'content-type': 'application/json', 'dangerouslyAllowBrowser': 'true' },

                body: JSON.stringify({

                    model: 'claude-3-5-sonnet-20240620', max_tokens: 1500,

                    messages: [{ role: 'user', content: prompt + " Respond ONLY with JSON." }]

                })

            });

            const data = await resp.json();

            return JSON.parse(data.content[0].text).opportunities;

        } else {

            const resp = await fetch('https://api.openai.com/v1/chat/completions', {

                method: 'POST',

                headers: { 'Authorization': `Bearer ${key}`, 'Content-Type': 'application/json' },

                body: JSON.stringify({

                    model: 'gpt-4o', messages: [{role: 'system', content: 'You return JSON only.'}, {role: 'user', content: prompt}],

                    response_format: { type: "json_object" }

                })

            });

            const data = await resp.json();

            return JSON.parse(data.choices[0].message.content).opportunities;

        }

    }


    function renderResults(list) {

        const container = document.getElementById('results');

        container.innerHTML = '';

        list.forEach(item => {

            const div = document.createElement('div');

            div.className = 'card';

            div.innerHTML = `

                <div class="badge">${item.matchScore}% Match</div>

                <div class="card-title">${item.title}</div>

                <div class="card-meta">${item.type} • ${item.location}</div>

                <div class="card-desc">${item.why || item.description}</div>

                <a href="${item.link}" target="_blank" class="card-link">View Original Listing ↗</a>

            `;

            container.appendChild(div);

        });

    }
